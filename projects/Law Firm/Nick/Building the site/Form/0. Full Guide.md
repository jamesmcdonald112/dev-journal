# **3️⃣ Zod Schema**

  

contact.schema.ts

```
import { z } from "zod";
import { PRACTICE_AREAS } from "./config/practice-areas";

export const contactSchema = z.object({
  name: z.string().trim().min(2).max(100),

  phone: z.string().trim().min(5).max(30),

  email: z.string().trim().email().max(254),

  practiceArea: z.enum(PRACTICE_AREAS),

  message: z.string().trim().max(2000).optional(),

  website: z.string().trim().nullable().optional().transform(v => v ?? ""),
});

export type ContactFormData = z.infer<typeof contactSchema>;
```

What this guarantees:

- All strings trimmed
    
- Honeypot always string
    
- Enum validation strict
    
- Typed safe data

# **5️⃣ Astro Action**

  

Responsibilities only:

- Origin check
    
- Parse input
    
- Honeypot block
    
- Call service
    
- Return success
    

  

actions/sendContact.ts

```
import { ActionError } from "astro:actions";
import { contactSchema } from "../contact.schema";
import { deliverContact } from "../service/deliverContact";
import { getAllowedOrigins } from "@/features/shared/config/allowed-origins";

export async function sendContactHandler(input: unknown, context: any) {
  const allowedOrigins = getAllowedOrigins();

  if (allowedOrigins.size > 0) {
    const origin = context?.request.headers.get("origin");
    if (!origin || !allowedOrigins.has(origin)) {
      throw new ActionError({ code: "FORBIDDEN", message: "Forbidden" });
    }
  }

  const parsed = contactSchema.parse(input);

  if (parsed.website.trim()) {
    return { success: true, message: "OK" };
  }

  await deliverContact(parsed);

  return { success: true, message: "Enquiry received." };
}
```

```
import { Resend } from "resend";
import type { ContactFormData } from "../contact.schema";

function escapeHtml(str: string) {
  return str
    .replace(/&/g, "&amp;")
    .replace(/</g, "&lt;")
    .replace(/>/g, "&gt;")
    .replace(/"/g, "&quot;");
}

export async function deliverContact(input: ContactFormData) {
  const apiKey = import.meta.env.RESEND_API_KEY;
  const toEmail = import.meta.env.CONTACT_TO_EMAIL;

  if (!apiKey) throw new Error("Missing RESEND_API_KEY");
  if (!toEmail) throw new Error("Missing CONTACT_TO_EMAIL");

  const resend = new Resend(apiKey);

  const subject = `New enquiry – ${escapeHtml(input.practiceArea)}`
    .replace(/[\r\n]+/g, " ");

  await resend.emails.send({
    from: "Mary Molloy Solicitor <info@yourdomain.ie>",
    to: toEmail.split(","),
    replyTo: input.email,
    subject,
    html: `
      <h2>New Enquiry</h2>
      <p><strong>Name:</strong> ${escapeHtml(input.name)}</p>
      <p><strong>Email:</strong> ${escapeHtml(input.email)}</p>
      <p><strong>Phone:</strong> ${escapeHtml(input.phone)}</p>
      <p><strong>Practice Area:</strong> ${escapeHtml(input.practiceArea)}</p>
      <p><strong>Message:</strong><br/>
        ${input.message
          ? escapeHtml(input.message).replace(/\n/g, "<br/>")
          : "—"}
      </p>
    `,
  });
}
```
